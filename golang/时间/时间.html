func main(){
  t := "2020-11-12 23:00"
  t = strings.Split(t," ")[0]
  s := strings.Split(t,"-")
  y,_ := strconv.Atoi(s[0])
  m,_ := strconv.Atoi(strings.TrimPrefix(s[1],"0"))
  d,_ := strconv.Atoi(strings.TrimPrefix(s[2],"0"))


  n:= time.Date(y,time.Month(m),d,0,0,0,0,time.Local)
  nt := time.Now()
  if int(nt.Sub(n).Hours()) > 48 {
    fmt.Println("干活")
  }
}
转为2006-01-02格式
func main() {
  // 定义日期字符串和对应的格式
  dateString := "6. Februar 2024"
  dateFormat := "2. January 2006"

  // 解析日期字符串为time.Time类型
  date, err := time.Parse(dateFormat, dateString)
  if err != nil {
    fmt.Println("日期解析错误:", err)
    return
  }

  // 将日期格式化为"2024-02-06"的字符串
  formattedDate := date.Format("2006-01-02")
  fmt.Println("转换后的日期:", formattedDate)
}
