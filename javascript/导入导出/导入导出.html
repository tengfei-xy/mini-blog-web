import
导入对象：对象、函数、类
基本语法
  • import 关键字是必须的。
  • defaultMember 是可选的，用于导入模块的默认导出。
  • { namedMember1, namedMember2 } 是可选的，用于导入模块的命名导出。
  • from 关键字是必须的。
  • "module-name" 是要导入的模块的名称。
import [defaultMember, { namedMember1, namedMember2 }] from "module-name";
示例
基本示例
import defaultExport from "module-name";
import * as name from "module-name";
import { export } from "module-name";
import { export as alias } from "module-name";
import { export1 , export2 } from "module-name";
import { foo , bar } from "module-name/path/to/specific/un-exported/file";
import { export1 , export2 as alias2 , [...] } from "module-name";
import defaultExport, { export [ , [...] ] } from "module-name";
import defaultExport, * as name from "module-name";
import "module-name";
var promise = import("module-name");//这是一个处于第三阶段的提案。
导入所有
import * as obj from "module-name";

// 使用 obj 对象
const name = obj.name;
const age = obj.age;
console.log(name); // "John Doe"
console.log(age); // 30
重命名导入
import { sum as total, average as avg } from "utils";

// 使用 total 和 avg 函数
const total = sum(1, 2, 3);
const avg = average(1, 2, 3);
console.log(total); // 6
console.log(avg); // 2
导入多个模块
import add from "math";
import { sum, average } from "utils";

// 使用 add、sum 和 average 函数
const result = add(1, 2);
const total = sum(1, 2, 3);
const avg = average(1, 2, 3);
console.log(result); // 3
console.log(total); // 6
console.log(avg); // 2
动态导入
const module = await import("module-name");

// 使用 module 对象
const add = module.add;
const result = add(1, 2);
console.log(result); // 3

export * from 'xxxx'
