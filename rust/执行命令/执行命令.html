use std::process::Command;

fn main() {
    // 要执行的PowerShell命令
    let command = "Write-Host 'Hello, World!'";

    // 创建一个PowerShell进程
    let output = Command::new("powershell")
        .arg("-Command")
        .arg(command)
        .output()
        .expect("无法执行命令");

    ，噜噜噜噜lllllllllllllllllllllllllllllllllllllllllllllllllllllllllll// 检查命令是否成功执行
    if output.status.success() {
        // 获取命令的输出结果
        let stdout = String::from_utf8_lossy(&output.stdout);
        println!("命令输出：{}", stdout);
    } else {
        // 获取命令的错误信息
        let stderr = String::from_utf8_lossy(&output.stderr);
        println!("命令执行出错：{}", stderr);
    }
}
