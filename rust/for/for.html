序列循环
    for i in 1..=5 {
        println!("{}", i);
    }
    
    for _ in 0..10 {
        // ...
    }
不转移所有权的循环
for item in &container {
  // ...
}
在循环中修改元素
for item in &mut collection {
  // ...
}
int遍历到0
let n = 9;

for i in (0..=n).rev() {
   println!("{}", i);
}
循环中获取索引
    let a = [4, 3, 2, 1];
    // `.iter()` 方法把 `a` 数组变成一个迭代器
    for (i, v) in a.iter().enumerate() {
        println!("第{}个元素是{}", i + 1, v);
    }
