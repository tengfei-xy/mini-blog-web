pub fn uncomparse_x(gzip_file: &str, output_dir: &str) -> io::Result<()> {
    let compressed_file_path = Path::new(gzip_file);
    let output_folder_path = Path::new(output_dir);
    // 创建输出文件夹（如果不存在）
    fs::create_dir_all(&output_folder_path)?;

    // 打开压缩文件
    let compressed_file = File::open(&compressed_file_path)?;
    let mut gz_decoder = GzDecoder::new(compressed_file);

    // 读取并解压缩文件
    let mut buffer = Vec::new();
    gz_decoder.read_to_end(&mut buffer)?;

    // 递归创建目录结构并写入解压缩的文件
    let mut tar = tar::Archive::new(&*buffer);
    tar.unpack(output_folder_path)
}
