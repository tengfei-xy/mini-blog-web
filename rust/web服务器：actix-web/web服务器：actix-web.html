示例
请求参数解析为json
use actix_web::{web, App, HttpServer, Responder, HttpResponse};
use serde::{Deserialize, Serialize};
#[derive(Debug,Deserialize,Serialize)]
struct UserRequest {
    username: String,
    password: String,
}

async fn login(req_json: web::Json<UserRequest>) -> impl Responder {
    // 使用 Json 结构体来提取 JSON 参数
    let user = req_json.into_inner();

    // 将 user 返回给客户端
    HttpResponse::Ok().json(user)
}

#[actix_web::main]
async fn main() -> std::io::Result<()> {

    HttpServer::new(|| {
        App::new()
            .route("/api/v1/login", web::post().to(login))
    })
    .bind("127.0.0.1:16825")? // 监听所有IP和端口8000
    .run()
    .await
}
在数据池中共享变量
共享结构体
use actix_web::{
    web::{self, Data},
    App, HttpResponse, HttpServer, Responder,
};
struct Dynu{
    a_recode: String,
}

async fn index (data: Data<Dynu>) ->impl Responder{
    HttpResponse::Ok().body(data.a_recode.clone())

}


#[actix_web::main]
async fn main() -> std::io::Result<()> {
    let target_ip = "translate.google.com";

    let share_data = Dynu::build(target_ip.to_string()).await.unwrap();

    HttpServer::new(move || {
        App::new()
            .app_data(web::Data::new(Dynu{
                a_recode: share_data.a_recode.clone(),
            }))
            .route("/", web::get().to(index))
    })
    .bind("127.0.0.1:16825")?
    .run()
    .await
}
在actix_web中使用mysql
说明：在 Rust 中，可以使用 mysql 库来连接 MySQL 数据库。在 Actix Web 服务器中，可以使用 Data 结构来存储数据库连接池，并将其注入到任何请求处理程序中。以下是使用 mysql 和 Actix Web 进行数据库查询的步骤：
添加依赖项
cargo add mysql
创建数据库连接池
use mysql::Pool;

fn init_mysql() -> Pool {
    let username = "";
    let password = "";
    let host = "";
    let database_name = "";   
    Pool::new(format!("mysql://{}:{}@{}/{}",username,password,host,database_name).as_str()).unwrap()
}
创建 Data 结构
use actix_web::web::Data;

struct MysqlData {
    pool: Data<Pool>,
}
将数据库连接池注入到请求处理程序中
use actix_web::{get, web, App, HttpServer, web:Data};

#[get("/")]
async fn index(data: Data<MysqlData>) -> impl Responder {
    let mut conn = data.pool.get_conn().unwrap();
    let results: std::prelude::v1::Result<Vec<Row>, Error> =
        conn.query("SELECT username from account");

    // ...

    Ok("Hello, world!")
}

fn main() {

     HttpServer::new(|| {
        App::new()
            .app_data(Data::new(init_mysql().clone()))
            .route("/api/v1/login", web::post().to(login))
            .route("/api/v1/sigup", web::post().to(sigup))
    })
    .bind("localhost:8080")
    .unwrap()
    .run()
    .unwrap();
}
在上述代码中，我们首先添加了 mysql 和 actix-web 依赖项。然后，我们创建了一个 Pool 对象来存储数据库连接池。接下来，我们创建了一个 Data 结构来存储数据库连接池。最后，我们将数据库连接池注入到请求处理程序中。
在请求处理程序中，我们可以使用 get_conn() 方法从连接池中获取一个连接。然后，我们可以使用 query() 方法执行数据库查询。
