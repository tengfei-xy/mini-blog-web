程序设置
http代理
use reqwest::{header,Proxy,ClientBuilder};

#[tokio::main]
async fn main() -> Result<(), reqwest::Error> {
    // 创建一个 ClientBuilder
    let client_builder = ClientBuilder::new();

    // 设置代理服务器
    let proxy = Proxy::http("http://proxy.example.com:8080")?;
    let client_builder = client_builder.proxy(proxy);

    // 构建客户端
    let client = client_builder.build()?;

    // 发送请求
    let response = client.get("https://www.example.com").send().await?;

    // 处理响应
    println!("Response: {:?}", response);

    Ok(())
}
