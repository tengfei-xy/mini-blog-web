clap
首先，将以下依赖项添加到你的 Cargo.toml 文件中：
[dependencies]
clap = "2.33.0"
下面是使用 clap 库的示例，我们定义了两个命令行参数 --input 和 --output，并使用 .value_name() 方法为它们指定了参数名。通过 .takes_value(true) 方法，我们告诉 clap 这些参数需要一个值。
use clap::{App, Arg};

fn main() {
    let matches = App::new("MyApp")
        .version("1.0")
        .author("Your Name")
        .about("Description of your app")
        .arg(
            Arg::with_name("input")
                .short("i")
                .long("input")
                .value_name("FILE")
                .help("Sets the input file")
                .takes_value(true),
        )
        .arg(
            Arg::with_name("output")
                .short("o")
                .long("output")
                .value_name("FILE")
                .help("Sets the output file")
                .takes_value(true),
        )
        .get_matches();

    if let Some(input) = matches.value_of("input") {
        println!("Input file: {}", input);
    }

    if let Some(output) = matches.value_of("output") {
        println!("Output file: {}", output);
    }
}
structopt
添加依赖
首先，将以下依赖项添加到你的 Cargo.toml 文件中：
cargo add structopt
全局访问
use std::sync::OnceLock;
use structopt::StructOpt;
#[derive(Debug, StructOpt)]
struct Opt {
    #[structopt(short = "f", long = "file", help = "指定文件名")]
    filename: String,

    #[structopt(short = "a", long = "output-full", help = "输出完整地址")]
    of: bool,
}
static OPT: OnceLock<Opt> = OnceLock::new();

fn init() {
    std::thread::spawn(|| {
         OPT.get_or_init(|| {
            Opt::from_args()
        });
    }).join().unwrap();
}
fn main(){
    init();
    println!("{}",OPT.get().unwrap().filename.as_str());
}
