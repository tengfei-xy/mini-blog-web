MySQL 5.7.25安装

#下载 带boost的mysql5.7
wget https://cdn.mysql.com/archives/mysql-5.7/mysql-boost-5.7.25.tar.gz

# centos
sudo yum -y install gcc gcc-c++ ncurses ncurses-devel bison libgcrypt perl make cmake bison-devel libaio openssl*

# ubutnu 
apt-get install cmake libncurses5-dev  bison g++

# 查看所有编译选项:cmake -LH
cmake . -DWITH_BOOST=~/mysql-5.7.25/boost/boost_1_59_0 -DCMAKE_INSTALL_PREFIX=/usr/local/services/mysql-5.7.25 -DMYSQL_UNIX_ADDR=/tmp/mysql.sock -DMYSQL_TCP_PORT=3306  -DWITH_INNOBASE_STORAGE_ENGINE=1 -DWITH_PARTITION_STORAGE_ENGINE=1 -DWITH_FEDERATED_STORAGE_ENGINE=1 -DWITH_BLACKHOLE_STORAGE_ENGINE=1 -DWITH_MYISAM_STORAGE_ENGINE=1 -DENABLED_LOCAL_INFILE=1 -DENABLE_DTRACE=0 -DDEFAULT_CHARSET=utf8mb4 -DDEFAULT_COLLATION=utf8mb4_general_ci -DWITH_EMBEDDED_SERVER=OFF

make -j `grep processor /proc/cpuinfo  | wc -l`
sudo make install
cd /usr/local/services/mysql-5.7.25
sudo groupadd mysql && sudo useradd -r -g mysql -s /bin/false mysql
sudo mkdir data logs etc 
sudo touch logs/mysql-error.log
sudo chown -R mysql:mysql ../mysql-5.7.25
sudo chmod -R 755 ../mysql-5.7.25
sudo bin/mysqld --initialize --user=mysql --basedir=/usr/local/services/mysql-5.7.25 --datadir=/usr/local/services/mysql-5.7.25/data
