macOS
下载
wget https://cdn.mysql.com/archives/mysql-5.7/mysql-boost-5.7.43.tar.gz
解压
tar xf mysql-boost-5.7.43.tar.gz
cmake
# 注意指定boost目录和安装目录
cmake . -DWITH_BOOST=/Users/melta/Downloads/mysql-5.7.43/boost -DCMAKE_INSTALL_PREFIX=/Users/melta/services/mysql-5.7.43 -DMYSQL_USER=mysql -DMYSQL_UNIX_ADDR=/Users/melta/services/mysql-5.7.43/mysql.sock -DMYSQL_TCP_PORT=3306  -DWITH_INNOBASE_STORAGE_ENGINE=1 -DWITH_PARTITION_STORAGE_ENGINE=1 -DWITH_FEDERATED_STORAGE_ENGINE=1 -DWITH_BLACKHOLE_STORAGE_ENGINE=1 -DWITH_MYISAM_STORAGE_ENGINE=1 -DENABLED_LOCAL_INFILE=1 -DENABLE_DTRACE=0 -DDEFAULT_CHARSET=utf8mb4 -DDEFAULT_COLLATION=utf8mb4_general_ci -DWITH_EMBEDDED_SERVER=OFF
make
make -j `sysctl hw.physicalcpu | grep "[[:digit:]]" -o ` && make install

cd /Users/melta/Downloads/mysql-5.7.43
mkdir data logs etc 
touch logs/mysql-error.log
bin/mysqld --initialize-insecure --user=melta --basedir=/Users/melta/services/mysql-5.7.43 --datadir=/Users/melta/services/mysql-5.7.43/data
