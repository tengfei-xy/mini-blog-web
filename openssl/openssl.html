编译为共享库
centos7
openssl-1.1.1s
wget https://www.openssl.org/source/openssl-1.1.1s.tar.gz --no-check-certificate
tar zxvf openssl-1.1.1s.tar.gz && cd openssl-1.1.1s
./config shared --prefix=/tmp/ssl
make -j `grep processor /proc/cpuinfo  | wc -l`
make install
openssl 3.2.1
安装依赖
yum -y install perl-IPC-Cmd perl-CPAN
安装模块
cpan -i Text::Template
下载
wget https://github.com/openssl/openssl/releases/download/openssl-3.2.1/openssl-3.2.1.tar.gz
解压
tar xf penssl-3.2.1.tar.gz
配置
生成静态库（例如libcrypto.3.so）
./Configure --prefix=/usr/local/services/openssl-3.2.1 --openssldir=/usr/local/services/openssl-3.2.1
编译
make -j `grep processor /proc/cpuinfo  | wc -l
安装
make install
macos
openssl 3.2.1
下载
wget https://github.com/openssl/openssl/releases/download/openssl-3.2.1/openssl-3.2.1.tar.gz
解压
tar xf penssl-3.2.1.tar.gz
配置
生成动态库（例如libcrypto.3.dylib）
./Configure --prefix=/Users/melta/Documents/source/lib/openssl-3.2.1
安装
make -j `sysctl hw.physicalcpu | grep "[[:digit:]]" -o`
make install
额外：导出变量
export OPENSSL_DIR=/Users/melta/Documents/source/lib/openssl-3.2.1
export OPENSSL_INCLUDE_DIR=$OPENSSL_DIR/include 
export OPENSSL_LIB_DIR=$OPENSSL_DIR/lib
